// --- COMPLIANCE & COOKIE LOGIC --- const cookieBanner = document.getElementById('cookie-banner'); const adSlot = document.getElementById('library-ad-slot'); const adElement = document.getElementById('library-ad-ins'); let adRequested = false; function hasAcceptedCookies() { return localStorage.getItem('cookieConsent') === 'accepted'; } function updateAdVisibility() { const currentSiteContent = document.getElementById('site-content'); const showAd = hasAcceptedCookies() && currentSiteContent.style.display !== 'none'; adSlot.classList.toggle('hidden', !showAd); if (!showAd) { (adsbygoogle = window.adsbygoogle || []).pauseAdRequests = 1; return; } (adsbygoogle = window.adsbygoogle || []).pauseAdRequests = 0; if (!adRequested) { try { (adsbygoogle = window.adsbygoogle || []).push({}); adRequested = true; adElement.dataset.adLoaded = 'true'; } catch (e) { console.warn('AdSense not ready yet', e); } } } const userConsent = localStorage.getItem('cookieConsent'); if (!userConsent) { cookieBanner.classList.add('visible'); (adsbygoogle = window.adsbygoogle || []).pauseAdRequests = 1; } else { updateAdVisibility(); } document.getElementById('accept-cookies').onclick = () => { localStorage.setItem('cookieConsent', 'accepted'); cookieBanner.classList.remove('visible'); updateAdVisibility(); }; document.getElementById('decline-cookies').onclick = () => { localStorage.setItem('cookieConsent', 'declined'); cookieBanner.classList.remove('visible'); updateAdVisibility(); };
